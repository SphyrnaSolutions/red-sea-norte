   ```typescript
   function mapStreamField(blocks: WagtailStreamFieldBlock[]): Block[] {
     return blocks.map(block => {

       if (block.type === 'rich_text' && typeof block.value === 'string') {
   ```

The blog listing currently loads full BlogPost[] including body (29 blocks per post x 25 posts = 391KB wasted). The listing only needs: slug, title, excerpt, hero (image, alt), category (name, color), readTime, publishedAt, author.name.

1. **Create a BlogPostListing type** in `src/lib/mock-data/types.ts`:

2. **Add a lightweight fetcher** in `src/lib/wagtail/fetchers.ts`:
   ```typescript
   export async function getAllBlogPostsListing(config?: FetchConfig): Promise<BlogPostListing[]> {
     const pages = await getPages<WagtailBlogPostPage>(
       'blog.BlogPostPage',
       { fields: 'title,excerpt,published_at,read_time,author,category,hero_image,hero_alt' },
       config
     )
     return pages
       .map(page => {
         const full = mapBlogPost(page)
         // eslint-disable-next-line @typescript-eslint/no-unused-vars
         const { body, ...listing } = full
         return listing
       })
       .sort((a, b) => {
         const aTime = Date.parse(a.publishedAt || '')
         const bTime = Date.parse(b.publishedAt || '')
         if (Number.isNaN(aTime) && Number.isNaN(bTime)) return 0
         if (Number.isNaN(aTime)) return 1
         if (Number.isNaN(bTime)) return -1
         return bTime - aTime
       })
   }
   ```

   NOTE: The Wagtail API `fields` parameter accepts comma-separated field names. By requesting specific fields instead of `*`, the API response will not include the `body` StreamField, dramatically reducing payload size. If the Wagtail API does not support field-level selection for StreamFields (test this), fall back to fetching all fields and stripping body in the mapper:
   ```typescript
   // Fallback approach if Wagtail ignores fields param for StreamFields
   const pages = await getPages<WagtailBlogPostPage>('blog.BlogPostPage', undefined, config)
   return pages.map(page => {
     const { body, ...listing } = mapBlogPost(page)
     return listing
   }).sort(...)
   ```
