**E. handleSubmit update (in both):**
Replace the simulated/TODO submit with:
```typescript
const handleSubmit = async (e: React.FormEvent) => {
  e.preventDefault()
  setSubmitError(null)
  if (!validateForm()) return

  setIsSubmitting(true)
  try {
    const response = await fetch("/api/lead", {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({
        name: formData.name,
        email: formData.email,
        phone: formData.phone,
        certificationLevel: formData.certificationLevel,
        preferredMonth: formData.preferredMonth,
        message: formData.message || undefined,
        whatsappConsent: whatsappConsent,
      }),
    })
    if (!response.ok) {
      const data = await response.json().catch(() => null)
      throw new Error(data?.error || "Error al enviar el formulario")
    }
    setShowSuccess(true)
    // For modal: auto-close after 3 seconds (existing behavior)
    // For inline: just show success, no auto-close
  } catch (err) {
    setSubmitError(err instanceof Error ? err.message : "Error inesperado. Intentalo de nuevo.")
  } finally {
    setIsSubmitting(false)
  }
}
```
